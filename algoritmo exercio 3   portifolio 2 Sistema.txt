algoritmo "Sistema de senha"

   senha_correta, senha_digitada : caractere

inicio
   senha_correta <- "senha1234567891011121314"

   escreval("-sistema de login-")
   escreva("Digite a senha para entrar: ")
   leia(senha_digitada)

   escreval()
   escreval("----")


   enquanto senha_digitada <> senha_correta faca
      escreval("Erro: Senha esta incorreta! Por favor Tente novamente.")
      escreval("-----")
      escreva("Digite a senha: ")
      leia(senha_digitada)
      
   fimenquanto

   escreval("---------")
   escreval("Seja Bem-vindo! O Acesso esta liberado.")
